ALTER  proc [dbo].[select_NH]
as
select nhomhang.MaNH, TenNH ,count(tenHH) as [Số lượng hàng] 
from [dbo].[nhomhang], hanghoa
where nhomhang.maNH = hanghoa.manh
group by nhomhang.maNH, TenNH
